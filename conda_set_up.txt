# ==== 1. Cài đặt Miniconda & Cấu hình Hệ thống ====
rm -rf /kaggle/working/miniconda3
cd /kaggle/working
wget -O Miniconda.sh https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh
bash Miniconda.sh -b -p /kaggle/working/miniconda3

# Thêm PATH và Load Conda
export PATH="/kaggle/working/miniconda3/bin:$PATH"
source /kaggle/working/miniconda3/etc/profile.d/conda.sh
echo 'export PATH="/kaggle/working/miniconda3/bin:$PATH"' >> ~/.bashrc
echo 'source /kaggle/working/miniconda3/etc/profile.d/conda.sh' >> ~/.bashrc

# Cài đặt Driver/Utils cần thiết
sudo apt install nvidia-utils-515 -y

# ==== 2. Tải Mã nguồn & Tạo Môi trường ====
git clone https://github.com/ducthinh4477/QHTD-CLIP
conda create -n QHTD python=3.10 -y 
conda activate QHTD

# ==== 3. Cài đặt thư viện (PyTorch & Dependencies) ====
# Cài đặt PyTorch CUDA 12.1 [cite: 648, 650]
pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu121

# Cài đặt các thư viện lõi của DeepfakeBench 
cd /kaggle/working/QHTD-CLIP/DeepfakeBench
bash install.sh

# Cài đặt bổ sung các thư viện thiếu phát sinh trong quá trình chạy 
pip install tensorboard future scikit-learn transformers filterpy albumentations
pip install timm opencv-python-headless
pip install numpy==1.23.5  # Bản ổn định tránh lỗi 'bool8' [cite: 650]

# Cài đặt dlib (khuyến khích dùng bản build sẵn từ conda-forge) 
conda install -c conda-forge dlib -y

# ==== 4. Cấu hình Dữ liệu (Symbolic Links) ====
# Liên kết thư mục JSON từ Input vào đúng vị trí Preprocessing 
rm -rf /kaggle/working/QHTD-CLIP/DeepfakeBench/preprocessing/dataset_json
ln -s /kaggle/input/datasets/xuanhuydinh/deepfakebench/dataset_json /kaggle/working/QHTD-CLIP/DeepfakeBench/preprocessing/dataset_json

# Liên kết dữ liệu hình ảnh (RGB) 
ln -s /kaggle/input/datasets/xuanhuydinh/deepfakebench/rgb /kaggle/working/rgb

# ==== 5. Kiểm tra & Bắt đầu Train ====
# Kiểm tra xem detector đã được đăng ký chưa 
python -c "from training.detectors import DETECTOR; print('effort_asy' in DETECTOR.data)"
